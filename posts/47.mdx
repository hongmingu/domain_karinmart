heroku create your-app-name

# 실제 예제
## 간단한 CRUD 애플리케이션 만들기
간단한 CRUD 애플리케이션을 만들려면 Sequel, SQLite와 같은 데이터베이스를 사용해 모델을 생성할 수 있다. 
이런 방법이 쉽고 간편하므로 추천하는 것 뿐이므로 만약 익숙하다면 튜토리얼 수준에서도 mysql을 postgresql등을 사용해도 무방하다.

```
require 'roda'
require 'sequel'

DB = Sequel.sqlite # SQLite를 사용하는 메모리 DB

class User < Sequel::Model(DB[:users])
end

class App < Roda
  plugin :json

  route do |r|
    r.on 'users' do
      # CREATE
      r.post do
        User.create(r.params)
      end

      # READ
      r.get do
        User.all
      end

      r.on ':id' do |id|
        user = User[id]

        # UPDATE
        r.put do
          user.update(r.params)
        end

        # DELETE
        r.delete do
          user.delete
        end
      end
    end
  end
end
```
